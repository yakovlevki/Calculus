\section{Лекция 3}
\subsection{Мера Жордана в $\R^k$}
\begin{definition}
    Если множество $Q$ раскладывается в объединение непересекающихся брусов $Q=\bigsqcup\limits_{i=1}^n \Pi_i$, то оно называется элементарным.
\end{definition}
% \begin{definition}
%     Выберем семейство элементарных множеств $\Pi_i$. Тогда семейство множеств вида $\bigsqcup\limits_{i=1}^n \Pi_i$, замкнутое относительно пересечения и симметрической разности, называется кольцом множеств.
% \end{definition}
\begin{definition}
    Пусть $\Phi$ -- ограниченное множество в $\R^k$. Тогда 
    \begin{enumerate}
        \item Если $\Pi$ --- брус, то определим $\mu(\Pi)=V(\Pi)$ и если $\Pi_i$ --- непересекающиеся брусы, то по определению положим 
        \[\mu\left(\ \bigsqcup\limits_{i=1}^n \Pi_i\right)=\sum\limits_{i=1}^{n}\mu(\Pi_i)\] 
        \item Внешней мерой Жордана называется число
            \[\mu^*(\Phi)=\inf \{\mu(P): P \text{ --- элементарное множество, такое, что } \Phi\subset P\}\]
        \item Внутренней мерой Жордана называется число
            \[\mu_*(\Phi)=\sup \{\mu(Q): Q \text{ --- элементарное множество, такое, что } Q\subset\Phi\}\]
        \item Если $\mu^*(\Phi)=\mu_*(\Phi)$, то назовем число 
            \[\mu(\Phi)=\mu^*(\Phi)=\mu_*(\Phi)\]
            мерой Жордана, а $\Phi$ измеримым по Жордану.
    \end{enumerate}
\end{definition}
\begin{definition}
    Функция 
    \[\chi_{\Phi}(x)=\begin{cases}
        1,\ x\in \Phi,\\
        0,\ x\not\in\Phi
    \end{cases}\]        
    называется индикатором множества $\Phi$.
\end{definition}
\begin{theorem} [Критерий измеримости по Жордану в терминах кратного интеграла] \label{kritizmvint} $\\$
    Множество $\Phi$ измеримо по Жордану $\Leftrightarrow$ по любому брусу $\Pi$ такому, что $\Phi\subseteq \Pi$, существует интеграл от $\chi_{\Phi}$ по $\Pi$. При этом
    \[\mu(\Phi)=\idotsint\limits_{\Pi} \chi_{\Phi}(x_1,\dots,x_k)\ dx_1\dots dx_k\]
\end{theorem}
\begin{proof}\tab
    \begin{itemize}
        \item[$(\Rightarrow)$:] Пусть $\mu_*(\Phi)=\mu^*(\Phi)=\mu(\Phi)$. Тогда по свойству точных граней\\
        $\forall \epsilon>0\ \exists\ P_{\epsilon},\ Q_{\epsilon},\ Q_{\epsilon}\subset \Phi\subset P_{\epsilon}$:
        \[\mu(P_{\epsilon})<\mu(\Phi)+\frac{\epsilon}{2},\ \ \mu(Q_{\epsilon})>\mu(\Phi)-\frac{\epsilon}{2}\]
        \[P_{\epsilon}=\bigsqcup\limits_i \Pi_i,\ Q_{\epsilon}=\bigsqcup\limits_j \Pi_j'\]
        можно считать, что $P_{\epsilon}$ лежат внутри бруса $\Pi$, если это не так, то, не увеличив оценку, заменим $P_{\epsilon}$ на
        \[\widehat{P}_{\epsilon}=\bigsqcup\limits_{i} (\Pi_i\cap \Pi)\]
        продолжив все грани $\Pi_i$ и $\Pi_j'$, получим разбиение $T=\{\delta_j\}$ бруса $\Pi$, при этом поскольку
        \[\overline{\overline{s}}(\chi_{\Phi},T)=\sum\limits_{j}\inf\limits_{x\in \Delta_j}\chi_{\Phi}(x)\cdot |\Delta_j|,\ \ \underline{\underline{S}}(\chi_{\Phi},T)=\sum\limits_{j}\sup\limits_{x\in \Delta_j}\chi_{\Phi}(x)\cdot |\Delta_j|\]
        то $\overline{\overline{s}}(\chi_{\Phi},T)$ состоит из элементарных множеств, которые полностью содержатся в $\Phi$, а $\underline{\underline{S}}(\chi_{\Phi},T)$ из элементарных множеств, которые пересекаются с $\Phi$ хотя бы по одной точке. Значит 
        %еще бы пояснить
        \[\overline{\overline{s}}(\chi_{\Phi}, T)\geq \mu(Q_{\epsilon}),\ \ \underline{\underline{S}}(\chi_{\Phi}, T)\leq \mu(P_{\epsilon})\]
        таким образом
        \[\Omega(\chi_{\Phi}, T)=\underline{\underline{S}}(\chi_{\Phi}, T)-\overline{\overline{s}}(\chi_{\Phi}, T)\leq \mu(P_{\epsilon})-\mu(Q_{\epsilon})<\frac{\epsilon}{2}+\frac{\epsilon}{2}=\epsilon\]
        значит существует
        \[I=\idotsint\limits_{\Pi} \chi_{\Phi}(x_1,\dots,x_k)\ dx_1\dots dx_k\]
        Причем
        \[\mu(\Phi)-\frac{\epsilon}{2}<\mu(Q_{\epsilon})\leq \overline{\overline{s}}(\chi_{\phi}, T)\leq I\leq \underline{\underline{S}}(\chi_{\phi}, T)\leq \mu(P_{\epsilon})<\mu(\Phi)+\frac{\epsilon}{2}\]
        Итак
        \[|I-\mu(\Phi)|<\frac{\epsilon}{2}\]
        Из произвольности $\epsilon$ получим, что
        \[I=\mu(\Phi)\]
        \item[$(\Leftarrow)$:] Пусть существует 
        \[I=\idotsint\limits_{\Pi} \chi_{\Phi}(x_1,\dots,x_k)\ dx_1\dots dx_k\] 
        По критерию Дарбу $\forall \epsilon>0\ \exists\ T_{\epsilon}=\{\Delta_j\}$ --- разбиение бруса $\Pi$ такое, что
        % \[\Omega(\chi_{\Phi}, T)=\sum\limits_{j}(\sup\limits_{\Delta_j}\chi_{\Phi}-\inf\limits_{\Delta_j}\chi_{\Phi})\cdot |\Delta_j|<\epsilon\]
        \[\Omega(\chi_{\Phi}, T)=\underline{\underline{S}}(\chi_{\Phi},T_{\epsilon})-\overline{\overline{s}}(\chi_{\Phi}, T_{\epsilon})<\epsilon\]
        Заметим, что $\Delta p$ делятся на группы:
        \begin{enumerate}
            \item $\sup\limits_{x\in \Delta_j} \chi_{\Phi}(x)=\inf\limits_{x\in \Delta_j} \chi_{\Phi}(x)=1$\ \ ($\Delta_j\subset \Phi$)
            \item $\sup\limits_{x\in \Delta_j} \chi_{\Phi}(x)=1,\ \inf\limits_{x\in \Delta_j} \chi_{\Phi}(x)=0$\ \ ($\Delta_j\cap \partial\Phi\neq \emptyset$)
            \item $\sup\limits_{x\in \Delta_j} \chi_{\Phi}(x)=\inf\limits_{x\in \Delta_j} \chi_{\Phi}(x)=0$\ \ ($\Delta_j\cap \Phi=\emptyset$)
        \end{enumerate}
        Определим $Q_{\epsilon}$ как объединение брусов из первой группы, а $P_{\epsilon}$ как объединение брусов из первой и второй группы. Заметим, что $\overline{\overline{s}}(\chi_{\Phi}, T_{\epsilon})$ является суммой мер брусов из первой группы, а $\underline{\underline{S}}(\chi_{\Phi}, T_{\epsilon})$ --- суммой мер брусов из первой и второй группы. Значит
        \[\overline{\overline{s}}(\chi_{\Phi}, T_{\epsilon})=\mu(Q_{\epsilon}),\ \ \underline{\underline{S}}(\chi_{\Phi}, T_{\epsilon})= \mu(P_{\epsilon}) \eqno(*)\]
        Таким образом
        % \[\Omega(\chi_{\Phi}, T)=\mu(P_{\epsilon})-\mu(Q_{\epsilon})\]
        \[\mu^*(\Phi)-\mu_*(\Phi)\leq \mu(P_{\epsilon})-\mu(Q_{\epsilon})=\Omega(\chi_{\Phi}, T_{\epsilon})<\epsilon\]
        Известно, что
        \[\overline{\overline{s}}(\chi_{\Phi}, T_{\epsilon})\leq I\leq \underline{\underline{S}}(\chi_{\Phi}, T_{\epsilon})\]
        \[\mu(Q_{\epsilon})\leq \mu(\Phi)\leq \mu(P_{\epsilon})\]
        комбинируя эти результаты и используя соотношение $(*)$, получим
        \[|I-\mu(\Phi)|\leq \Omega(\chi_{\Phi}, T_{\epsilon})<\epsilon\]
    \end{itemize}
\end{proof}
\begin{consequense}
    Пусть $\Phi$ измеримо по Жордану и $\Phi\subset \Pi_1,\ \Phi\subset \Pi_2$. Тогда
    \[\idotsint\limits_{\Pi_1} \chi_{\Phi}(x_1,\dots,x_k)\ dx_1\dots dx_k=\idotsint\limits_{\Pi_2} \chi_{\Phi}(x_1,\dots,x_k)\ dx_1\dots dx_k\]
\end{consequense}
\subsection{Первое доказательство критерия измеримости по Жордану}
\begin{theorem}[Критерий измеримости по Жордану] $\\$
    Следующие условия эквивалентны:
    \begin{enumerate}
        \item $\Phi$ измеримо по Жордану
        \item $\forall \epsilon>0\ \exists$ элементарные множества $P_{\epsilon}, Q_{\epsilon}:\ Q_{\epsilon}\subset \Phi\subset P_{\epsilon}$ такие, что
        \[\mu(P_{\epsilon})-\mu(Q_{\epsilon})<\epsilon\]
        \item $\mu(\partial \Phi)=0$
    \end{enumerate}
\end{theorem}
\begin{proof} Докажем по схеме $(1)\Rightarrow(2) \Rightarrow(3) \Rightarrow (1)$
    \begin{enumerate}
        \item $(1) \Rightarrow (2)$:\\
        Пусть $\Phi$ измеримо по Жордану. Тогда по теореме \ref{kritizmvint} существует интеграл
        \[\idotsint\limits_{\Pi} \chi_{\Phi}(x_1,\dots,x_k)\ dx_1\dots dx_k\]
        Дальнейшие выкладки полностью повторяют доказательство обратной импликации предыдущей теоремы:
        По критерию Дарбу $\forall \epsilon>0\ \exists\ T_{\epsilon}=\{\Delta_j\}$ --- разбиение бруса $\Pi$ такое, что
        % \[\Omega(\chi_{\Phi}, T)=\sum\limits_{j}(\sup\limits_{\Delta_j}\chi_{\Phi}-\inf\limits_{\Delta_j}\chi_{\Phi})\cdot |\Delta_j|<\epsilon\]
        \[\Omega(\chi_{\Phi}, T)=\underline{\underline{S}}(\chi_{\Phi},T_{\epsilon})-\overline{\overline{s}}(\chi_{\Phi}, T_{\epsilon})<\epsilon\]
        Заметим, что $\Delta p$ делятся на группы:
        \begin{enumerate}
            \item $\sup\limits_{x\in \Delta_j} \chi_{\Phi}(x)=\inf\limits_{x\in \Delta_j} \chi_{\Phi}(x)=1$\ \ ($\Delta_j\subset \Phi$)
            \item $\sup\limits_{x\in \Delta_j} \chi_{\Phi}(x)=1,\ \inf\limits_{x\in \Delta_j} \chi_{\Phi}(x)=0$\ \ ($\Delta_j\cap \partial\Phi\neq \emptyset$)
            \item $\sup\limits_{x\in \Delta_j} \chi_{\Phi}(x)=\inf\limits_{x\in \Delta_j} \chi_{\Phi}(x)=0$\ \ ($\Delta_j\cap \Phi=\emptyset$)
        \end{enumerate}
        Определим $Q_{\epsilon}$ как объединение брусов из первой группы, а $P_{\epsilon}$ как объединение брусов из первой и второй группы. Заметим, что $\overline{\overline{s}}(\chi_{\Phi}, T_{\epsilon})$ является суммой мер брусов из первой группы, а $\underline{\underline{S}}(\chi_{\Phi}, T_{\epsilon})$ --- суммой мер брусов из первой и второй группы. Значит
        \[\overline{\overline{s}}(\chi_{\Phi}, T_{\epsilon})=\mu(Q_{\epsilon}),\ \ \underline{\underline{S}}(\chi_{\Phi}, T_{\epsilon})= \mu(P_{\epsilon}) \eqno(*)\]
        Таким образом
        % \[\Omega(\chi_{\Phi}, T)=\mu(P_{\epsilon})-\mu(Q_{\epsilon})\]
        \[\mu(P_{\epsilon})-\mu(Q_{\epsilon})=\Omega(\chi_{\Phi}, T_{\epsilon})<\epsilon\]
        \item $(2) \Rightarrow (3)$:\\
        Пусть 
        \[\mu(P_{\epsilon})-\mu(Q_{\epsilon})<\epsilon\]
        Заметим, что $P_{\epsilon}\setminus Q_{\epsilon}$ является элементарным множеством. Тогда
        \[\partial \Phi \subset \overline{P_{\epsilon}\setminus Q_{\epsilon}}\]
        при этом
        \[\mu\left(\overline{P_{\epsilon}\setminus Q_{\epsilon}}\right)=\mu(P_{\epsilon})-\mu(Q_{\epsilon})<\epsilon\]
        \item $(3) \Rightarrow (1)$:\\
        Пусть $\mu(\partial \Phi)=0$. Тогда существует элементарное множество $R_{\epsilon}$, содержащее $\partial \Phi$, такое, что $\mu(R_{\epsilon})<\epsilon$.
        Продолжим грани брусов в его составе и получим разбиение $T$ бруса $\Pi$, который содержит $\Phi$. Поскольку
        \[\overline{\overline{s}}(\chi_{\Phi},T)=\sum\limits_{p}\inf\limits_{x\in \Delta p}\chi_{\Phi}(x)|\Delta p|,\ \ \underline{\underline{S}}(\chi_{\Phi},T)=\sum\limits_{p}\sup\limits_{x\in \Delta p}\chi_{\Phi}(x)|\Delta p|\]
        то $\overline{\overline{s}}(\chi_{\Phi},T)$ состоит из элементарных множеств, которые полностью содержатся в $\Phi$, а $\underline{\underline{S}}(\chi_{\Phi},T)$ из элементарных множеств, которые пересекаются с $\Phi$ хотя бы по одной точке. Отсюда понятно, что
        \[\Omega(\chi_{\Phi}, T)\leq \mu(R_{\epsilon})<\epsilon\]
        Значит, по предыдущей теореме $\Phi$ измеримо по Жордану.
    \end{enumerate}
\end{proof}
\begin{example}[Неизмеримого по Жордану множества]$\\$
    Пусть $\Q_{[0,1]}=\Q\cap [0,1]$. Заметим, что $(\Q_{[0,1]})^k\subset ([0,1])^k$ и\\
    $\partial (\Q_{[0,1]})^k=([0,1])^k$. Таким образом $\mu(\partial(\Q_{[0,1]})^k)=1\neq 0$, значит, $(Q_{[0,1]})^k$ неизмеримо по Жордану.
\end{example}
\begin{example}[Ковер Серпинского] $\\$
    Рассмотрим единичный квадрат на плоскости. Разделим его в заданном соотношении на 9 прямоугольников прямыми, параллельными сторонам, и вырежем центральный прямоугольник. Далее на каждом шаге все имеющиеся прямоугольники также разрезаем на 9 частей и выкидываем центральную. Фигура, полученная с помощью такого бесконечного процесса, называется ковром (квадратом) Серпинского.\\
    Внутренняя мера ковра Серпинского равна нулю, так как вырезанные части в нём всюду плотны. Однако можно подобрать параметры так, чтобы внешняя мера была положительной. Например, пусть на $k$-м шаге из прямоугольника вырезается подобный прямоугольник с площадью в $3^{2k}$ раз меньшей площади исходного. Тогда внешняя мера будет равна 
    \[1\cdot(1-\frac{1}{3^2})\cdot(1-\frac{1}{3^4})... = \prod\limits_{i=1}^\infty(1-\frac{1}{3^k}) > 0\]
    т.к. ряд $\sum \frac{1}{3^{2i}}$ сходится. 
\end{example}
\begin{example}(Подграфик непрерывной на измеримом компакте функции)\\
    Пусть $\overline{G}$ -- измеримый компакт в $\R^{k-1}, f(x_1,...,x_{k-1})$ -- непрерывная на $\overline{G}$ функция. Рассмотрим $\Phi = \{(x_1,...,x_k) : (x_1,...,x_{k-1}) \in \overline{G}, 0 \leqslant x_k \leqslant f(x_1,...,x_{k-1})\}$.\\
    Покажем, что $\Phi$ измеримо. $\partial\Phi$ состоит из трёх частей:
    \begin{enumerate}
        \item Графика $x_k = f(x_1,...,x_{k-1})$ -- из курса действительного анализа известно, что в $\R^k$ мера графика непрерывной на $R^{k-1}$-мерном компакте функции равна 0;
        \item Проекции графика $\overline{G} \times \{0\}$ -- его мера равна $\mu(\overline{G}) \cdot 0 = 0$;
        \item Боковой границы -- она содержится на границе цилиндра с основанием $\overline{G}$ и высотой $\max\limits_{\overline{G}} f$, то есть её мера тажке нулевая.
    \end{enumerate}
    Значит, $\mu(\partial\Phi) = 0$, а отсюда $\Phi$ измеримо по Жордану.
\end{example}
