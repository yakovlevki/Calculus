\section{Лекция 9}
\noindent Скалярное произведение векторов $\overline{x}$ и $\overline{y}$ будем обозначать $\langle \overline{x},\ \overline{y} \rangle$
\begin{lemma}\label{l1}
    Пусть $\Phi$ --- выпуклое подмножество в $\R^k$, и в некоторой окрестности $U(\Phi)$ определена функция $\phi: U(\Phi)\to \R,\ \phi\in \mathcal{C}^1(\Phi)$. Тогда $\forall\ \overline{x}, \overline{y}\in \Phi\\ \exists\ \overline{\xi}\in [\overline{x}, \overline{y}]$:
    \[\phi(\overline{x})-\phi(\overline{y})=\langle \grad \phi(\overline{\xi}),\ \overline{x}-\overline{y} \rangle\]
\end{lemma}
\begin{proof}
    Поскольку $\Phi$ --- выпукла, то весь отрезок 
    \[[\overline{x}, \overline{y}]=\{\overline{y}+t(\overline{x}-\overline{y}):\ t\in [0,1]\}\] 
    содержится в $\Phi$. Тогда для $t\in [0,1]$ введем 
    \[h(t):=\phi(\overline{y}+t\cdot (\overline{x}-\overline{y}))\]
    \begin{multline*}
        h^{\prime}(t)=(\phi(y_1+t(x_1-y_1),\dots,y_k+t(x_k-y_k)))'=\\
        =\sum\limits_{j=1}^{k}\phi_j'(\overline{y}+t(\overline{x}-\overline{y}))\cdot (x_j-y_j)=\langle \grad \phi(\overline{y}+t(\overline{x}-\overline{y})), \overline{x}-\overline{y} \rangle
    \end{multline*}
    Тогда по теореме Лагранжа $\exists\ \theta\in (0,1)$ такое, что
    \[h(1)-h(0)=h'(\theta)(1-0)\]
    а значит
    \[\phi(\overline{x})-\phi(\overline{y})=\langle \grad \phi(\overline{y}+\theta(\overline{x}-\overline{y})), \overline{x}-\overline{y} \rangle\]
    где $\overline{\xi}=\overline{y}+\theta(\overline{x}-\overline{y})\in [\overline{x}, \overline{y}]$.
\end{proof}
\noindent Вспомним что такое норма в $\R^k$. Пусть $\overline{x}=(x_1,\dots,x_k)$. Евклидова норма задается
\[\|x\|_2=\sqrt{x_1^2+\dots x_k^2}\]
\begin{exercise} 
    Доказать, что при $p\geq 1$
    \[\|\overline{x}\|_p=(|x_1|^p+\dots+|x_k|^p)^{\frac{1}{p}}\]
    является нормой.
\end{exercise}
В частности, норма 
\[\|\overline{x}\|_1=(|x_1|^p+\dots+|x_k|^p)^{\frac{1}{p}}\] 
называется Манхэттенской. Также можно ввести норму 
\[\|\overline{x}\|_{\infty}=\max\limits_{j} |x_j|\]
\begin{lemma} [О липшицевости на компакте] $\\$
    Пусть $\Phi$ --- выпуклый компакт в $\R^k$, и в некоторой окрестности $U(\Phi)$ определена функция $\phi: U(\Phi)\to \R^m,\ \phi\in \mathcal{C}^1(\Phi)$. Тогда $\exists\ M>0\ \forall\ \overline{x}, \overline{y}\in \Phi$:
    \[\|\phi(\overline{x})-\phi(\overline{y})\|_{\R^m}\leq M\cdot \|\overline{x}-\overline{y}\|_{\R^k}\]
\end{lemma}
\begin{proof}
\begin{multline*}
    \sqrt{(\phi_1(\overline{x})-\phi_1(\overline{y}))^2+\dots+(\phi_m(\overline{x})-\phi_m(\overline{y}))^2}\leq \sum\limits_{i=1}^{m}|\phi_i(\overline{x})-\phi_i(\overline{y})|\overset{(1)}{=}\\
    \overset{(1)}{=}\sum\limits_{i=1}^{n}|\langle \grad \phi_i(\overline{\xi_i}),\ \overline{x}-\overline{y}\rangle|\overset{(2)}{\leq} \sum\limits_{i=1}^{m}\|\grad \phi_i(\overline{\xi}_i)\|_{\R^k}\cdot \|\overline{x}-\overline{y}|\|_{\R^k}
\end{multline*}
(1): По лемме \ref{l1}.\\
(2): По неравенству Коши-Буняковского.\\
Поскольку $\phi\in \mathcal{C}^1(\Phi)$, то каждая $\frac{\partial \phi_i}{\partial x_j}\in \mathcal{C}(\Phi)$. Тогда $\exists\ M_i>0$:
\[\|\grad(\phi_i(\overline{\xi}_i))\|_{\R^k}=\sqrt{\sum\limits_{j=1}^{k}\left(\frac{\partial \phi_i}{\partial x_j}(\overline{\xi}_i)\right)^2}\leq M_i\]
Таким образом
\[M=\sum\limits_{i=1}^{m}M_i\]
\end{proof}
\begin{reminder}
    Пусть $\Omega\subset \R^k,\ \phi: \Omega\to \R^m$. Тогда матрица
    \[\begin{pmatrix}
        \frac{\partial \phi_1}{\partial x_1} & \frac{\partial \phi_1}{\partial x_2} & \dots & \frac{\partial \phi_1}{\partial x_k}\\
        \frac{\partial \phi_2}{\partial x_1} & \frac{\partial \phi_2}{\partial x_2} & \dots & \frac{\partial \phi_2}{\partial x_k}\\
        \vdots & \vdots & \null & \vdots\\
        \frac{\partial \phi_m}{\partial x_1} & \frac{\partial \phi_m}{\partial x_2} & \dots & \frac{\partial \phi_m}{\partial x_k}\\
    \end{pmatrix}\]
    называется матрицей Якоби. Если $m=k$, то определитель этой матрицы называют Якобианом, его обозначают $J$.
    Пусть $\phi$ --- дифференцируема в точке $\overline{x}_0$. Тогда
    \[\phi(\overline{x}_0+\Delta \overline{x})-\phi(\overline{x}_0)=J(\overline{x}_0)\cdot \Delta \overline{x}+\alpha(\Delta \overline{x})\cdot \|\Delta \overline{x}\|_{\R^k}\]
    где $\alpha(\Delta \overline{x})=\bar{\bar{o}}(1)$.
\end{reminder}

\begin{lemma} [О липшивости производной на компакте] \label{l9.3} $\\$
    Пусть $\Phi$ --- выпуклый компакт в $\R^k$, и в некоторой окрестности $U(\Phi)$ определена функция $\phi: U(\Phi)\to \R^m,\ \phi\in \mathcal{C}^1(\Phi)$. Тогда $\exists\ \omega(t): \R_+\to \R_+$,\\ причем $\lim\limits_{t\to 0+}\omega(t)=0$ а также $\forall\ \overline{x},\ \overline{x}+\Delta\overline{x}\in \Phi$:
    \[\|\phi(\overline{x}+\Delta\overline{x})-\phi(\overline{x})-J(\overline{x})\cdot \Delta\overline{x}\|_{\R^m}\leq \omega(\|\Delta\overline{x}\|_{\R_k})\cdot \|\Delta\overline{x}\|_{\R^k}\] 
\end{lemma}
\begin{proof}
    Применим лемму \ref{l1} к каждой координате:
    \[\phi_i(\overline{x}-\Delta\overline{x})-\phi(\overline{x})=(\langle \grad \phi_1(\overline{\xi}_1), \Delta \overline{x} \rangle, \dots, \langle \grad \phi_m(\overline{\xi}_m), \Delta \overline{x} \rangle)\]
    при этом
    \[J(\overline{x})\cdot \Delta \overline{x}=(\langle \grad \phi_1(\overline{x}), \Delta \overline{x} \rangle, \dots, \langle \grad \phi_m(\overline{x}), \Delta \overline{x} \rangle)\]
    значит
    \begin{multline*}
        \phi(\overline{x}+\Delta\overline{x})-\phi(\overline{x})-J(\overline{x})\cdot \Delta\overline{x}=\\
        =(\langle \grad \phi_1(\overline{\xi}_1)-\grad \phi_1(\overline{x}), \Delta \overline{x} \rangle, \dots, \langle \grad \phi_1(\overline{\xi}_m)-\grad \phi_1(\overline{x}), \Delta \overline{x} \rangle)
    \end{multline*}
    Тогда 
    \begin{multline*}
        \|\phi(\overline{x}+\Delta \overline{x})-\phi(\overline{x})-J(\overline{x})\cdot \Delta\overline{x}\|\leq\\
        \leq \sum\limits_{i=1}^{m}|\langle \grad \phi_i(\overline{\xi}_i)-\grad \phi_i(\overline{x}), \Delta\overline{x} \rangle|\leq\\
        \leq \left(\sum\limits_{i=1}^{m}\|\grad \phi_i(\overline{\xi}_i)-\grad \phi_i(\overline{x})\|\right)\cdot \|\Delta \overline{x}\|
    \end{multline*}
    Введем 
    \[\omega_{ij}(\delta):=\sup\left\{\ \left|\frac{\partial \phi_i}{\partial x_j}(\overline{x})-\frac{\partial \phi_i}{\partial x_j}(\overline{y})\right|:\ \overline{x}, \overline{y}\in \Phi,\ \|\overline{x}-\overline{y}\|<\delta\right\}\]
    Поскольку $\phi\in \mathcal{C}^1(\Phi)$, то все частные производные непрерывны, значит $\omega_{ij}(\delta)\to 0$ при $\delta\to 0+$.
    Теперь определим
    \[\omega(\delta)=\sum\limits_{i=1}^{m}\sum\limits_{j=1}^{k}\omega_{ij}(\delta)\]
    и заметим что $\omega(\delta)\to 0$ при $\delta\to 0+$. Ограничим первый множитель
    \[\sum\limits_{i=1}^{m}\|\grad \phi_i(\overline{\xi}_i)-\grad \phi_i(\overline{x})\|\leq \sum\limits_{i=1}^{m}\sum\limits_{j=1}^{k}\left|\frac{\partial \phi_i}{\partial x_j}(\overline{x})-\frac{\partial \phi_i}{\partial x_j}(\overline{\xi}_i)\right|\leq \omega(\|\Delta \overline{x}\|)\]
    Итак
    \[\|\phi(\overline{x}+\Delta \overline{x})-\phi(\overline{x})-J(\overline{x})\cdot \Delta\overline{x}\|\leq \omega(\|\Delta \overline{x}\|)\cdot \|\Delta \overline{x}\|\]
    и $\omega$ обладает нужными свойствами.
\end{proof}
